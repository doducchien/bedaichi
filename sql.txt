create database hrm;
use hrm;
create table user(
	email varchar(200),
    fullName nvarchar(200),
    phoneNumber varchar(20),
    birthday varchar(30),
    sex tinyint,
    type varchar(20),
    password varchar(100)
);

create table staff(
	email varchar(200),
    fullName nvarchar(200),
    phoneNumber varchar(20),
    birthday varchar(30),
    sex tinyint,
    image varchar(300),
    department varchar(100),
    position varchar(100),
    joinDay varchar(30),
    leftDay varchar(30),
    status tinyint,
    note text
);

create table timeKeeping(
	email varchar(200),
    time varchar(30)
    -- numberWorkDays tinyint,
--     numberOvertimeDays tinyint
);

create table unitSalary(
	email varchar(200),
	basicSalary varchar(20),
    overtimeSalary varchar(20),
    allowance varchar(20),
    performanceBonus varchar(20),
    attendanceBonus varchar(20),
    completedBonus varchar(20),
    awarenessBonus varchar(20),
    note text
);

create table totalSalary(
	email varchar(200),
    time varchar(30),
    basicSalary varchar(200),
    overtimeSalary varchar(200),
    allowance varchar(20),
    performanceBonus varchar(20),
    attendanceBonus varchar(20),
    completedBonus varchar(20),
    awarenessBonus varchar(20),
    totalSalary varchar(20),
    advanceSalary varchar(20),
    note text
);

create table product(
	id varchar(30),
    name nvarchar(300),
    value varchar(20),
    time varchar(30),
    isComplete boolean
);

create table staffProduct(
	id varchar(30),
    email varchar(200),

create table customer(

	email varchar(200),

    leaveWithPermission tinyint,

    leaveSpecial tinyint,
    forgotCheckin tinyint,
    note text
);

create table regime(
	email varchar(200),
    time varchar(30),
    nameRegime text,

    ispaid boolean,
    note text
);



create table ordered(
	email varchar(200),
    time varchar(30),
    id varchar(30),
    price varchar(20),

	note text
